The "Deferred" object

  • Has a ".resolve()" method for success.
  • Has a ".reject()" for failure.

function getUser(){
   var userData = {
     "name" : "John Doe",
     "email" : "jd_1234@email.com",
     "phone" : "555-555-5555"
   };

   var DFD = $.Deferred();

   DFD.resolve(userData);  //This example is synchronous, it doesn't matter.
   // -or-
   DFD.reject("User not known");
   
   return DFD.promise();
}
